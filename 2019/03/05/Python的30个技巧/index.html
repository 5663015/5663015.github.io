25.使用slots减少内存开支

import sys
# 原始类
class FileSystem(object):
	def __init__(self, files, folders, devices):
		self.files = files
		self.folder = folders
		self.devices = devices
		print(sys.getsizeof(FileSystem))
# 减少内存后
class FileSystem(object):
	__slots__ = ['files', 'folders', 'devices']
	def __init__(self, files, folders, devices):
		self.files = files
		self.folder = folders
		self.devices = devices
		print(sys.getsizeof(FileSystem))
